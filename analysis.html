echo_cohort <- master %>%
  filter(
    echo_ef_notavailable___999      == 0 |
    echo_late_notavailable___999    == 0 |
    echo_gls_notavailable___999     == 0 |
    echo_earatio_notavailable___999 == 0 |
    echo_eeratio_notavailable___999 == 0
  ) %>%
  # Harmonize echo_gls sign convention: some centers store negatives (the
  # standard convention), others store the absolute value. Take abs() once
  # here so every downstream summary (medians, abnormal flag, spline) uses
  # magnitudes only.
  mutate(echo_gls = abs(echo_gls))

preterli_echo <- echo_cohort %>%
  filter(days_to_echo < days_to_terli_initiation_day0) %>%
  mutate(
    # Abnormal-echo flags (1 = abnormal, 0 = normal, NA if measurement missing)
    echo_ef_abnormal        = if_else(is.na(echo_ef),
                                      NA_real_, as.numeric(echo_ef < 60)),
    echo_eeratio_abnormal   = if_else(is.na(echo_eeratio),
                                      NA_real_, as.numeric(echo_eeratio > 15)),
    # echo_gls is already stored as a magnitude (see harmonization in echo_cohort).
    echo_gls_abnormal       = if_else(is.na(echo_gls),
                                      NA_real_, as.numeric(echo_gls < 18)),
    echo_lateral_e_abnormal = if_else(is.na(echo_lateral_e),
                                      NA_real_, as.numeric(echo_lateral_e < 10)),
    # TR is unit-aware: < 100 treated as m/s (cutoff 2.8); >= 100 as cm/s (cutoff 280).
    echo_tr_abnormal        = if_else(is.na(echo_tr),
                                      NA_real_,
                                      as.numeric(if_else(echo_tr < 100,
                                                         echo_tr > 2.8,
                                                         echo_tr > 280))),
    echo_lavi_abnormal      = if_else(is.na(echo_lavi),
                                      NA_real_, as.numeric(echo_lavi > 34))
  )

timing_counts <- echo_cohort %>%
  mutate(timing = case_when(
    days_to_echo <  days_to_terli_initiation_day0 ~ "1. echo before terli",
    days_to_echo == days_to_terli_initiation_day0 ~ "2. echo same day as terli",
